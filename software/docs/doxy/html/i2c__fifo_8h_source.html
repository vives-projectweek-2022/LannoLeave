29static inline uint8_t i2c_read_byte(i2c_inst_t *i2c) {
30 i2c_hw_t *hw = i2c_get_hw(i2c);
31 assert(hw->status & I2C_IC_STATUS_RFNE_BITS); // Rx FIFO must not be empty
32 return (uint8_t)hw->data_cmd;
33}
34
43static inline void i2c_write_byte(i2c_inst_t *i2c, uint8_t value) {
44 i2c_hw_t *hw = i2c_get_hw(i2c);
45 assert(hw->status & I2C_IC_STATUS_TFNF_BITS); // Tx FIFO must not be full
46 hw->data_cmd = value;
47}
